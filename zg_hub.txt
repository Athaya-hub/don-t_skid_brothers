local autoTarget = false
local autoDelay = 0.2

local function getNearest()
   local closest, dist = nil, math.huge
   for _,v in game.Players:GetPlayers() do
      if v ~= plr and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
         local d = (plr.Character.HumanoidRootPart.Position - v.Character.HumanoidRootPart.Position).Magnitude
         if d < dist then
            dist = d
            closest = v
         end
      end
   end
   return closest
end

task.spawn(function()
   while true do
      if autoTarget then
         local target = getNearest()
         if target then
            local root = target.Character.HumanoidRootPart
            for i = 1, hitAmount do
               AcidRemote:FireServer(root.Position + Vector3.new(0,10,0), root.Position)
            end
         end
      end
      task.wait(autoDelay)
   end
end)

   Name = "Touch Target Acid",
   CurrentValue = false,
   Callback = function(Value)

         targetPlayer = nil
         if highlight then
            highlight:Destroy()
            highlight = nil
         end
      end
   end
})

AcidTab:CreateSlider({
   Name = "Hit Amount",
   Range = {1, 15},
   Increment = 1,
   CurrentValue = 1,
   Callback = function(Value)
      hitAmount = Value
   end
})

-- Detect tap / click
UIS.InputBegan:Connect(function(input)
   if not touchTargetEnabled then return end

   if input.UserInputType ~= Enum.UserInputType.Touch
   and input.UserInputType ~= Enum.UserInputType.MouseButton1 then
      return
   end

   local ray = camera:ScreenPointToRay(input.Position.X, input.Position.Y)
   local result = workspace:Raycast(ray.Origin, ray.Direction * 500)

   if result and result.Instance then
      local model = result.Instance:FindFirstAncestorOfClass("Model")
      local player = game.Players:GetPlayerFromCharacter(model)

      if player then
         targetPlayer = player

         -- remove old highlight
         if highlight then
            highlight:Destroy()
         end

         -- add highlight
         highlight = Instance.new("Highlight")
         highlight.Parent = model
         highlight.FillColor = Color3.fromRGB(0,255,0)
      end
   end
end)

-- Acid loop (TARGETED)
task.spawn(function()
   while true do
      if touchTargetEnabled and targetPlayer then
         if targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local root = targetPlayer.Character.HumanoidRootPart

            for i = 1, hitAmount do
   AcidRemote:FireServer(
      root.Position + Vector3.new(0,10,0),
      root.Position
   )
end
         end
      end

      task.wait(0.1)
   end
end)
